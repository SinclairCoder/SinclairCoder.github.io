## [HDU2032](http://acm.hdu.edu.cn/showproblem.php?pid=2032)
>还记得中学时候学过的杨辉三角吗？具体的定义这里不再描述，你可以参考以下的图形：
1
1 1
1 2 1
1 3 3 1
1 4 6 4 1
1 5 10 10 5 1

即元素值等于肩上的两个元素之和，每行第一个跟最后一个都为１。

<!-- more -->
````java
//解法１、二维数组
int main() 
{
	int n,a[30][30]={0};
	while(cin>> n)
	{
		for(int i=0;i<n;i++)
		{
			a[i][0]=1;a[i][i]=1;
		}
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<=i;j++)
			{
				if(j>0)
					a[i][j] = a[i-1][j] + a[i-1][j-1];
				cout << a[i][j];
				if(j<i) cout << " ";
			}
			cout << endl;
		}
		cout << endl;
	}
 	return 0;
}

//解法2、一维数组
int main() 
{
	int n;	
	while(cin>>n)
	{
		int a[40]={0},b[40]={0}; //此处会涉及到初始化问题，故须放到while循环里面
		a[1]=1;b[1]=1;
		for(int i=1;i<=n;i++) // 此处索引从1开始
		{
			for(int j=1;j<=i;j++)
			{
				a[j]=b[j]+b[j-1]; 	
			}
			for(int j=1;j<=i;j++)
			{
				b[j] = a[j] ; 
				cout<< b[j] ;
				if(j<i) cout << " ";
			}	
			cout << endl;
		}
		cout << endl;
		
	}
 	return 0;
}

//解法3、多项式　杨辉三角中每一行的值为C（ｎ－１，ｍ），其中n为行数，m为列数索引
int fun(int n,int m)
{
	int s1=1,j=1;
	for(int i=n;i>n-m;i--)
	{
		s1 =s1*i/j;
		j++;
	}
	return s1;
}
int main() 
{
	int n;
	while(cin>> n)
	{
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<=i;j++)
			{
				cout << fun(i,j);
				if(j<i)cout << " ";
			}
			cout << endl;
		} 
		cout << endl;
	}
 	return 0;
}

// 解法4、递推法，发现规律
int main() 
{
	int n;	
	while(cin>>n)
	{
		for(int i=1;i<=n;i++)
		{
			int b=1;
			for(int j=1;j<=i;j++)
			{
				cout << b;
				b = b*(i-j)/j;  // 下一个元素值=上一个*(行数-列数)/列数 
				if(j<i) cout << " ";
			
			}
			cout << endl;
		}
		cout << endl;
		
	}
 	return 0;
}
````
